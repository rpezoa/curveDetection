\begin{equation}
X[k]= \begin{cases}
 \sum_{n=0}^{N-1} x[n]W_{N}^{kn}, & 0\leq k \leq N-1, W_{k}=e^{-j\frac{2\pi}{N}} \\
 0, & \text{otherwise}
\end{cases}
\end{equation}



\begin{equation}
x[n]= \begin{cases}
 \frac{1}{N}\sum_{k=0}^{N-1} X[k]W_{N}^{-kn}, & 0\leq n \leq N-1 \\
 0, & \text{otherwise}
\end{cases}
\end{equation}


$$W_{N}=e^{-j \frac{ 2 i\pi}{N}}$$
